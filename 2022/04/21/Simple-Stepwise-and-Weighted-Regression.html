for i in range(0,len(num),1):
    x = pd.DataFrame(data[num[0:i+1]])
    y = data['price']
    model = sm.OLS(y, x).fit()

    print("With predictor {}".format(num[0:i+1]))
    print("R-squared = ", model.rsquared)
    print("Adj R-squared =", model.rsquared_adj)
    print("AIC =", model.aic)
    print("BIC =", model.bic)
    print("\n")
